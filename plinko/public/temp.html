let positions = [1, 1.5, 2, 1.5, 1, 1.5, 2, 2.5, 3, 2.5]; $(document).ready(function(){ let pos = $('#chip').position(); console.log(pos); $("#StartButton").click(function() { let slot = $('#SelectedSlot').val(); $('#chip').animate({top: pos.top = 0, left: pos.left = 50*(slot-1)}, 1); $("#chip").animate({top: pos.top = pos.top+50}, 500); for (let i = 1; i < positions.length; i++) { if(positions[i] > positions[i-1]) { $("#chip").animate({top: pos.top = pos.top+50, left: pos.left = pos.left+50}, 500); } else { $("#chip").animate({top: pos.top = pos.top+50, left: pos.left = pos.left-50}, 500); } } }); });
